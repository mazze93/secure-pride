# Secure Pride Extended v1 — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Secure Pride PKI trust auditor CLI — Security.framework scanners, filesystem key detection, optional Docker trust inspection, trust chain graph, and a tabbed local web dashboard with AUDIT FEED and TRUST GRAPH views.

**Architecture:** Swift CLI with ArgumentParser. Six scanner types: Certificate, Identity, Trust (always via Security.framework), Filesystem (always on), Docker (opt-in via `--include docker`). GraphBuilder resolves cert chains from warn/critical findings into nodes+edges. ScoringEngine applies rules per category. NWListener HTTP server serves `Dashboard/index.html` + `/api/scan.json`. Dashboard is a single HTML file with two tabs rendered in vanilla JS+SVG — no CDN libraries.

**Tech Stack:** Swift 5.9+, macOS 13+, Security.framework, Network.framework, Foundation, swift-argument-parser ~1.3.0. Zero third-party dependencies beyond ArgumentParser.

**Spec references:**
- `docs/secure-pride-v1-design.md` — full v1 spec (scoring rules, data model, CLI, dashboard server)
- This plan extends v1 with filesystem scanning, Docker inspection, and the trust graph

import Foundation

struct Finding: Codable, Equatable {
    let id: String
    let category: Category
    let name: String
    let detail: String
    let status: Status
    let score: Int
    let metadata: [String: String]
}

enum Category: String, Codable, Equatable {
    case certificate    = "certificate"
    case identity       = "identity"
    case trustSetting   = "trustSetting"
    case filesystemKey  = "filesystemKey"
    case dockerMount    = "dockerMount"
}

enum Status: String, Codable, Equatable {
    case ok       = "ok"
    case warn     = "warn"
    case critical = "critical"
}

// Standard metadata keys — use these strings as dictionary keys
// "algorithm"   e.g. "RSA 2048", "EC 256", "RSA 1024"
// "issuer"      subject summary of the issuing certificate
// "expiry"      ISO 8601 date, e.g. "2026-12-31T00:00:00Z"
// "keychain"    e.g. "login", "System", "System Roots"
// "domain"      trust domain: "user", "admin", "system"
// "policy"      trust policy: "trustAll", "ssl", "codeSign"
// "keyType"     e.g. "RSA", "EC", "Ed25519"
// "keySizeBits" e.g. "2048", "256"
// "path"        filesystem path for filesystemKey findings
// "permissions" octal permission string, e.g. "0644"
// "container"   Docker container ID for dockerMount findings
// "mountDest"   Docker mount destination path

// Sources/SecurePride/Models/TrustGraph.swift
import Foundation

struct TrustGraph: Codable, Equatable {
    let nodes: [TrustNode]
    let edges: [TrustEdge]

    static let empty = TrustGraph(nodes: [], edges: [])
}

struct TrustNode: Codable, Equatable {
    let id: String         // matches Finding.id or synthetic "SYN_{hash}"
    let label: String      // human-readable name
    let type: NodeType
    let status: Status     // ok/warn/critical for colour coding in UI
}

struct TrustEdge: Codable, Equatable {
    let from: String       // parent node ID (issuer)
    let to: String         // child node ID (subject)
}

enum NodeType: String, Codable, Equatable {
    case root         = "root"          // self-signed, no parent in graph
    case intermediate = "intermediate"  // has both parent and children
    case leaf         = "leaf"          // has parent, no children in graph
    case synthetic    = "synthetic"     // issuer not in findings; placeholder root
}
```

// Sources/SecurePride/Scanners/CertificateScanner.swift
import Foundation
import Security

struct CertificateScanner {

    static func scan() -> (findings: [Finding], errors: [ScanError]) {
        var findings: [Finding] = []
        var errors: [ScanError] = []

        let query: [String: Any] = [
            kSecClass as String:        kSecClassCertificate,
            kSecReturnRef as String:    true,
            kSecMatchLimit as String:   kSecMatchLimitAll
        ]

        var result: CFTypeRef?
        let status = SecItemCopyMatching(query as CFDictionary, &result)

        switch status {
        case errSecSuccess:
            guard let certs = result as? [SecCertificate] else { break }
            for (index, cert) in certs.enumerated() {
                findings.append(makeFinding(cert: cert, index: index))
            }
        case errSecItemNotFound:
            break
        case errSecAuthFailed, errSecInteractionNotAllowed:
            errors.append(ScanError(
                scanner: "CertificateScanner",
                message: "Keychain access denied",
                code: status
            ))
        default:
            errors.append(ScanError(
                scanner: "CertificateScanner",
                message: "SecItemCopyMatching failed",
                code: status
            ))
        }

        return (findings, errors)
    }

    // MARK: - Private

    private static func makeFinding(cert: SecCertificate, index: Int) -> Finding {
        let name = SecCertificateCopySubjectSummary(cert) as String? ?? "Unknown Certificate"
        let id   = String(format: "CERT_%03d", index + 1)
        var metadata: [String: String] = [:]

        // Key algorithm and size
        if let key = SecCertificateCopyKey(cert) {
            let attrs    = SecKeyCopyAttributes(key) as? [String: Any]
            let keyType  = attrs?[kSecAttrKeyType as String] as? String ?? ""
            let keySize  = attrs?[kSecAttrKeySizeInBits as String] as? Int ?? 0
            metadata["keyType"]     = humanKeyType(keyType)
            metadata["keySizeBits"] = "\(keySize)"
            metadata["algorithm"]   = humanAlgorithm(keyType: keyType, keySize: keySize)
        }

        // Expiry date
        if let expiry = copyExpiryDate(cert: cert) {
            metadata["expiry"] = ISO8601DateFormatter().string(from: expiry)
        }

        // Issuer
        metadata["issuer"] = copyIssuerSummary(cert: cert) ?? ""

        let (status, score) = classify(metadata: metadata)
        return Finding(
            id: id, category: .certificate, name: name,
            detail: buildDetail(status: status, metadata: metadata),
            status: status, score: score, metadata: metadata
        )
    }

    private static func copyExpiryDate(cert: SecCertificate) -> Date? {
        let oids = [kSecOIDX509V1ValidityNotAfter] as CFArray
        guard let values = SecCertificateCopyValues(cert, oids, nil) as? [String: Any],
              let dict   = values[kSecOIDX509V1ValidityNotAfter as String] as? [String: Any],
              let raw    = dict[kSecPropertyKeyValue as String]
        else { return nil }
        if let date  = raw as? Date   { return date }
        if let num   = raw as? NSNumber {
            return Date(timeIntervalSinceReferenceDate: num.doubleValue)
        }
        return nil
    }

    private static func copyIssuerSummary(cert: SecCertificate) -> String? {
        let oids = [kSecOIDX509V1IssuerName] as CFArray
        guard let values = SecCertificateCopyValues(cert, oids, nil) as? [String: Any],
              let dict   = values[kSecOIDX509V1IssuerName as String] as? [String: Any],
              let pairs  = dict[kSecPropertyKeyValue as String] as? [[String: Any]]
        else { return nil }

        for pair in pairs {
            let label = pair[kSecPropertyKeyLabel as String] as? String ?? ""
            let value = pair[kSecPropertyKeyValue as String] as? String ?? ""
            if label == "2.5.4.3" || label.contains("Common Name") { return value }
        }
        for pair in pairs {
            let label = pair[kSecPropertyKeyLabel as String] as? String ?? ""
            let value = pair[kSecPropertyKeyValue as String] as? String ?? ""
            if label == "2.5.4.10" || label.contains("Organization") { return value }
        }
        return nil
    }

    private static func humanKeyType(_ oid: String) -> String {
        switch oid {
        case kSecAttrKeyTypeRSA as String:              return "RSA"
        case kSecAttrKeyTypeEC as String,
             kSecAttrKeyTypeECSECPrimeRandom as String: return "EC"
        default:                                        return oid
        }
    }

    private static func humanAlgorithm(keyType: String, keySize: Int) -> String {
        let t = humanKeyType(keyType)
        return keySize > 0 ? "\(t) \(keySize)" : t
    }

    private static func classify(metadata: [String: String]) -> (Status, Int) {
        let algo = metadata["algorithm"] ?? ""
        if algo.contains("RSA 1024") || algo.contains("SHA1") { return (.critical, -20) }

        if let expiryStr = metadata["expiry"],
           let expiry    = ISO8601DateFormatter().date(from: expiryStr) {
            if expiry < Date()                                           { return (.critical, -15) }
            if expiry < Date().addingTimeInterval(30 * 24 * 3600)       { return (.warn, -10) }
        }
        return (.ok, 0)
    }

    private static func buildDetail(status: Status, metadata: [String: String]) -> String {
        let algo = metadata["algorithm"] ?? "unknown algorithm"
        switch status {
        case .critical where (metadata["algorithm"] ?? "").contains("RSA 1024"):
            return "Weak key: RSA 1024. Upgrade to RSA 2048 or EC 256."
        case .critical:
            return "Expired certificate. Renew immediately."
        case .warn:
            return "Certificate expires within 30 days. Schedule renewal."
        case .ok:
            return "Valid certificate (\(algo))."
        }
    }
}

// Sources/SecurePride/Scanners/IdentityScanner.swift
import Foundation
import Security

struct IdentityScanner {

    static func scan() -> (findings: [Finding], errors: [ScanError]) {
        var findings: [Finding] = []
        var errors:   [ScanError] = []

        let query: [String: Any] = [
            kSecClass as String:        kSecClassIdentity,
            kSecReturnRef as String:    true,
            kSecMatchLimit as String:   kSecMatchLimitAll
        ]

        var result: CFTypeRef?
        let status = SecItemCopyMatching(query as CFDictionary, &result)

        switch status {
        case errSecSuccess:
            guard let identities = result as? [SecIdentity] else { break }
            for (index, identity) in identities.enumerated() {
                findings.append(makeFinding(identity: identity, index: index))
            }
        case errSecItemNotFound:
            break
        case errSecAuthFailed, errSecInteractionNotAllowed:
            errors.append(ScanError(scanner: "IdentityScanner",
                                    message: "Keychain access denied", code: status))
        default:
            errors.append(ScanError(scanner: "IdentityScanner",
                                    message: "SecItemCopyMatching failed", code: status))
        }

        return (findings, errors)
    }

    private static func makeFinding(identity: SecIdentity, index: Int) -> Finding {
        let id = String(format: "IDENT_%03d", index + 1)
        var certRef: SecCertificate?
        SecIdentityCopyCertificate(identity, &certRef)

        let name = certRef.flatMap { SecCertificateCopySubjectSummary($0) as String? }
                   ?? "Unknown Identity"
        var metadata: [String: String] = [:]

        let identityType = classifyType(name: name)
        metadata["identityType"] = identityType

        if let cert = certRef, let key = SecCertificateCopyKey(cert) {
            let attrs   = SecKeyCopyAttributes(key) as? [String: Any]
            let keyType = attrs?[kSecAttrKeyType as String] as? String ?? ""
            let keySize = attrs?[kSecAttrKeySizeInBits as String] as? Int ?? 0
            metadata["keyType"]     = humanKeyType(keyType)
            metadata["keySizeBits"] = "\(keySize)"
        }

        let (status, score) = identityType == "unknown"
            ? (Status.critical, -25)
            : (Status.warn,     -5)

        return Finding(
            id: id, category: .identity, name: name,
            detail: buildDetail(identityType: identityType),
            status: status, score: score, metadata: metadata
        )
    }

    private static func classifyType(name: String) -> String {
        let lower = name.lowercased()
        if lower.contains("developer") || lower.contains("apple development") { return "developer" }
        if lower.contains("tls") || lower.contains("ssl") || lower.contains("localhost") { return "tls" }
        if lower.contains("ssh") { return "ssh" }
        return "unknown"
    }

    private static func buildDetail(identityType: String) -> String {
        switch identityType {
        case "developer": return "Apple developer identity (cert + private key). Verify this machine should hold signing keys."
        case "tls":       return "TLS server identity. Confirm it is expected on this machine."
        case "ssh":       return "SSH certificate identity stored in keychain."
        default:          return "Unknown identity type. Review and remove if not recognised."
        }
    }

    private static func humanKeyType(_ oid: String) -> String {
        switch oid {
        case kSecAttrKeyTypeRSA as String: return "RSA"
        case kSecAttrKeyTypeEC  as String: return "EC"
        default:                           return oid
        }
    }
}

// Sources/SecurePride/Scanners/TrustScanner.swift
import Foundation
import Security

struct TrustScanner {

    static func scan() -> (findings: [Finding], errors: [ScanError]) {
        var findings: [Finding] = []
        var errors:   [ScanError] = []
        var index = 1

        for (domain, name) in [(SecTrustSettingsDomain.user, "user"),
                                (SecTrustSettingsDomain.admin, "admin")] {
            var certArray: CFArray?
            let status = SecTrustSettingsCopyCertificates(domain, &certArray)

            switch status {
            case errSecSuccess:
                guard let certs = certArray as? [SecCertificate] else { continue }
                for cert in certs {
                    findings.append(makeFinding(cert: cert, domain: name, index: index))
                    index += 1
                }
            case errSecNoTrustSettings:
                continue
            default:
                errors.append(ScanError(scanner: "TrustScanner",
                                        message: "Failed for domain \(name)", code: status))
            }
        }
        return (findings, errors)
    }

    private static func makeFinding(cert: SecCertificate, domain: String, index: Int) -> Finding {
        let id   = String(format: "TRUST_%03d", index)
        let name = SecCertificateCopySubjectSummary(cert) as String? ?? "Unknown Trust Anchor"
        var metadata: [String: String] = ["domain": domain]

        var trustSettings: CFArray?
        // Query trust settings for the domain this cert came from
        let trustDomain: SecTrustSettingsDomain = domain == "admin" ? .admin : .user
        SecTrustSettingsCopyTrustSettings(cert, trustDomain, &trustSettings)
        if let settings = trustSettings as? [[String: Any]], !settings.isEmpty {
            let policies = settings.compactMap { s -> String? in
                (s[kSecTrustSettingsResult as String] as? NSNumber)
                    .flatMap { trustResultLabel($0.int32Value) }
            }
            metadata["policy"] = policies.joined(separator: ", ")
        }

        let (status, score) = classify(name: name, domain: domain)
        return Finding(
            id: id, category: .trustSetting, name: name,
            detail: buildDetail(status: status, name: name, domain: domain),
            status: status, score: score, metadata: metadata
        )
    }

    private static func classify(name: String, domain: String) -> (Status, Int) {
        if name.lowercased().contains("mkcert") { return (.warn, -15) }
        // Admin-domain additions are elevated-privilege installs — critical per spec
        if domain == "admin" { return (.critical, -25) }
        if domain == "user"  { return (.warn, -15) }
        return (.ok, 0)
    }

    private static func buildDetail(status: Status, name: String, domain: String) -> String {
        if name.lowercased().contains("mkcert") {
            return "mkcert root CA detected. The root key at ~/.local/share/mkcert/rootCA-key.pem can sign for any domain."
        }
        switch status {
        case .critical: return "Unknown root CA in \(domain) trust domain. Investigate."
        case .warn:     return "Custom certificate in \(domain) trust domain: \(name). Verify this is intentional."
        case .ok:       return "Standard trust anchor in \(domain) domain."
        }
    }

    private static func trustResultLabel(_ result: Int32) -> String? {
        switch result {
        case Int32(kSecTrustSettingsResultTrustRoot):   return "trustRoot"
        case Int32(kSecTrustSettingsResultTrustAsRoot): return "trustAsRoot"
        case Int32(kSecTrustSettingsResultDeny):        return "deny"
        default:                                         return nil
        }
    }
}

// Sources/SecurePride/Scanners/FilesystemScanner.swift
import Foundation

struct FilesystemScanner {

    private static let keyExtensions: Set<String> = ["key", "pem", "p12", "pfx", "jks", "der"]

    // Explicit sensitive hidden directories to scan even though .skipsHiddenFiles is set
    // for the general $HOME walk (avoids scanning all of .npm, .rbenv, etc.)
    private static let sensitiveHiddenDirs = [
        ".ssh",
        ".local/share/mkcert",  // mkcert root key lives here
        ".config/mkcert",
    ]

    static func scan() -> (findings: [Finding], errors: [ScanError]) {
        var findings: [Finding] = []
        var errors:   [ScanError] = []

        guard let home = ProcessInfo.processInfo.environment["HOME"] else {
            errors.append(ScanError(scanner: "FilesystemScanner",
                                    message: "$HOME not set", code: 0))
            return (findings, errors)
        }

        var index = 1
        let fm = FileManager.default

        // 1. General $HOME scan — skips hidden dirs (fast)
        let homeURL = URL(fileURLWithPath: home)
        if let enumerator = fm.enumerator(
            at: homeURL,
            includingPropertiesForKeys: [.isRegularFileKey],
            options: [.skipsHiddenFiles, .skipsPackageDescendants]
        ) {
            for case let url as URL in enumerator {
                if let f = checkFile(url: url, index: &index) { findings.append(f) }
            }
        } else {
            errors.append(ScanError(scanner: "FilesystemScanner",
                                    message: "Cannot enumerate \(home)", code: 0))
        }

        // 2. Explicit hidden dirs — targeted scan for known key locations
        for rel in sensitiveHiddenDirs {
            let dirURL = URL(fileURLWithPath: home).appendingPathComponent(rel)
            guard fm.fileExists(atPath: dirURL.path) else { continue }
            guard let enumerator = fm.enumerator(
                at: dirURL,
                includingPropertiesForKeys: [.isRegularFileKey],
                options: [.skipsPackageDescendants]
            ) else { continue }
            for case let url as URL in enumerator {
                if let f = checkFile(url: url, index: &index) { findings.append(f) }
            }
        }

        return (findings, errors)
    }

    // Returns a Finding if the URL is a private key file, nil otherwise.
    private static func checkFile(url: URL, index: inout Int) -> Finding? {
        let ext = url.pathExtension.lowercased()
        guard keyExtensions.contains(ext) else { return nil }
        guard looksLikePrivateKey(at: url) || ext == "key" || ext == "p12" || ext == "pfx" else { return nil }

        let path = url.path
        var metadata: [String: String] = ["path": path]
        let fm = FileManager.default

        if let perms = try? fm.attributesOfItem(atPath: path)[.posixPermissions] as? Int {
            metadata["permissions"] = String(perms, radix: 8)
        }

        let worldReadable = isWorldReadable(path: path)
        let inGitRepo     = path.contains("/.git/") || path.contains("/git/")
        let inDockerPath  = path.lowercased().contains("/docker") || path.lowercased().contains("/volumes")

        let (status, score, detail) = classify(
            path: path, isWorldReadable: worldReadable,
            isInGitRepo: inGitRepo, isInDockerPath: inDockerPath
        )

        defer { index += 1 }
        return Finding(
            id: String(format: "FILE_%03d", index),
            category: .filesystemKey,
            name: url.lastPathComponent,
            detail: detail,
            status: status, score: score, metadata: metadata
        )
    }

    // MARK: - Private

    private static func looksLikePrivateKey(at url: URL) -> Bool {
        guard let handle = try? FileHandle(forReadingFrom: url) else { return false }
        defer { try? handle.close() }
        let data: Data
        if #available(macOS 10.15.4, *) {
            data = (try? handle.read(upToCount: 512)) ?? Data()
        } else {
            data = handle.readData(ofLength: 512)
        }
        guard let header = String(data: data, encoding: .utf8) else { return false }
        return header.contains("PRIVATE KEY") || header.contains("RSA PRIVATE") || header.contains("EC PRIVATE")
    }

    private static func isWorldReadable(path: String) -> Bool {
        var st = stat()
        return stat(path, &st) == 0 && (st.st_mode & S_IROTH) != 0
    }

    private static func classify(
        path: String,
        isWorldReadable: Bool,
        isInGitRepo: Bool,
        isInDockerPath: Bool
    ) -> (Status, Int, String) {
        if isWorldReadable {
            return (.critical, -40, "World-readable private key at \(path). Run: chmod 600")
        }
        if isInGitRepo {
            return (.critical, -30, "Private key inside a git repository at \(path). Remove and rotate immediately.")
        }
        if isInDockerPath {
            return (.warn, -15, "Private key in Docker-related path at \(path). Verify container exposure.")
        }
        return (.warn, -10, "Private key file at \(path). Ensure owner-only permissions (chmod 600).")
    }
}

// Sources/SecurePride/Scanners/DockerScanner.swift
import Foundation

struct DockerScanner {

    private static let sensitiveDestPaths = [
        "/etc/ssl", "/etc/pki", "/certs", "/dev-pki",
        "/usr/local/share/ca-certificates", "/usr/share/ca-certificates"
    ]

    static func scan() -> (findings: [Finding], errors: [ScanError]) {
        var findings: [Finding] = []
        var errors:   [ScanError] = []

        guard dockerIsAvailable() else {
            errors.append(ScanError(
                scanner: "DockerScanner",
                message: "docker not available or daemon not running — skipped",
                code: 0
            ))
            return (findings, errors)
        }

        let (idsOut, _) = shell("docker ps -q 2>/dev/null")
        let ids = idsOut.split(separator: "\n").map(String.init).filter { !$0.isEmpty }
        guard !ids.isEmpty else { return (findings, errors) }

        var index = 1
        for id in ids {
            let (out, status) = shell("docker inspect \(id) 2>/dev/null")
            guard status == 0,
                  let data      = out.data(using: .utf8),
                  let json      = try? JSONSerialization.jsonObject(with: data) as? [[String: Any]],
                  let container = json.first
            else { continue }

            findings += inspectContainer(container: container, id: id, index: &index)
        }

        return (findings, errors)
    }

    private static func dockerIsAvailable() -> Bool {
        shell("docker info >/dev/null 2>&1").1 == 0
    }

    private static func inspectContainer(
        container: [String: Any],
        id: String,
        index: inout Int
    ) -> [Finding] {
        guard let mounts = container["Mounts"] as? [[String: Any]] else { return [] }

        let name = (container["Name"] as? String ?? id)
            .trimmingCharacters(in: CharacterSet(charactersIn: "/"))

        return mounts.compactMap { mount -> Finding? in
            let source = mount["Source"]      as? String ?? ""
            let dest   = mount["Destination"] as? String ?? ""
            guard sensitiveDestPaths.contains(where: { dest.hasPrefix($0) }) else { return nil }

            let (status, score) = dest.hasPrefix("/dev-pki")
                ? (Status.critical, -20)
                : (Status.warn,     -10)

            defer { index += 1 }
            return Finding(
                id: String(format: "DOCK_%03d", index),
                category: .dockerMount,
                name: "Container: \(name)",
                detail: "Container '\(name)' mounts '\(source)' to '\(dest)'. Verify PKI exposure.",
                status: status, score: score,
                metadata: ["container": id, "mountDest": dest, "mountSource": source]
            )
        }
    }

    @discardableResult
    private static func shell(_ command: String) -> (String, Int32) {
        let task = Process()
        task.executableURL = URL(fileURLWithPath: "/bin/bash")
        task.arguments = ["-c", command]
        let pipe = Pipe()
        task.standardOutput = pipe
        task.standardError  = Pipe()
        do { try task.run() } catch { return ("", -1) }
        task.waitUntilExit()
        let data = pipe.fileHandleForReading.readDataToEndOfFile()
        return (String(data: data, encoding: .utf8) ?? "", task.terminationStatus)
    }
}

// Sources/SecurePride/Graph/GraphBuilder.swift
import Foundation

struct GraphBuilder {

    /// Build a findings-anchored TrustGraph.
    /// Only certificate and trustSetting findings with warn/critical status are included.
    /// Walks issuer chains upward; creates synthetic placeholder nodes for unknown issuers.
    static func build(from findings: [Finding]) -> TrustGraph {
        let chainable = findings.filter {
            ($0.category == .certificate || $0.category == .trustSetting) &&
            $0.status != .ok
        }
        guard !chainable.isEmpty else { return .empty }

        // Name → first matching finding (for chain walking)
        var byName: [String: Finding] = [:]
        for f in findings where f.category == .certificate || f.category == .trustSetting {
            if byName[f.name] == nil { byName[f.name] = f }
        }

        var included = Set<String>()
        var nodes: [TrustNode] = []
        var edges: [TrustEdge] = []

        for finding in chainable {
            collectChain(finding: finding, byName: byName,
                         included: &included, nodes: &nodes, edges: &edges)
        }

        // Assign final node types from topology
        let childIDs  = Set(edges.map(\.to))
        let parentIDs = Set(edges.map(\.from))

        let typedNodes = nodes.map { node -> TrustNode in
            if node.type == .synthetic { return node }
            let isParent = parentIDs.contains(node.id)
            let isChild  = childIDs.contains(node.id)
            let type: NodeType = isParent && isChild  ? .intermediate
                               : isParent && !isChild ? .root
                               : !isParent && isChild ? .leaf
                               :                        .root   // isolated
            return TrustNode(id: node.id, label: node.label, type: type, status: node.status)
        }

        return TrustGraph(nodes: typedNodes, edges: edges)
    }

    // MARK: - Private

    private static func collectChain(
        finding: Finding,
        byName: [String: Finding],
        included: inout Set<String>,
        nodes: inout [TrustNode],
        edges: inout [TrustEdge]
    ) {
        guard !included.contains(finding.id) else { return }
        included.insert(finding.id)
        nodes.append(TrustNode(id: finding.id, label: finding.name,
                               type: .leaf, status: finding.status))

        let issuer = finding.metadata["issuer"] ?? ""
        if issuer.isEmpty || issuer == finding.name { return }  // self-signed

        if let parent = byName[issuer] {
            collectChain(finding: parent, byName: byName,
                         included: &included, nodes: &nodes, edges: &edges)
            edges.append(TrustEdge(from: parent.id, to: finding.id))
        } else {
            // Use stable FNV-1a hash — Swift's hashValue is randomised per-process
            let synID = "SYN_\(fnv1a(issuer))"
            if !included.contains(synID) {
                included.insert(synID)
                nodes.append(TrustNode(id: synID, label: issuer,
                                       type: .synthetic, status: .ok))
            }
            edges.append(TrustEdge(from: synID, to: finding.id))
        }
    }

    /// Deterministic FNV-1a 64-bit hash (no CryptoKit dependency).
    private static func fnv1a(_ s: String) -> UInt64 {
        var hash: UInt64 = 14_695_981_039_346_656_037
        for byte in s.utf8 {
            hash ^= UInt64(byte)
            hash = hash &* 1_099_511_628_211
        }
        return hash
    }
}

// Sources/SecurePride/Scoring/ScoringEngine.swift
import Foundation

struct ScoringEngine {

    /// Compute total score 0–100. Baseline 100; only warn/critical findings subtract.
    static func totalScore(from findings: [Finding]) -> Int {
        let deductions = findings.reduce(0) { $0 + $1.score }
        return max(0, min(100, 100 + deductions))
    }

    /// Build a ReportSummary from a findings list.
    static func summarize(findings: [Finding]) -> ReportSummary {
        ReportSummary(
            certificateCount:   findings.filter { $0.category == .certificate }.count,
            identityCount:      findings.filter { $0.category == .identity }.count,
            trustAnchorCount:   findings.filter { $0.category == .trustSetting }.count,
            filesystemKeyCount: findings.filter { $0.category == .filesystemKey }.count,
            dockerFindingCount: findings.filter { $0.category == .dockerMount }.count,
            criticalCount:      findings.filter { $0.status == .critical }.count,
            warnCount:          findings.filter { $0.status == .warn }.count
        )
    }

    /// Human-readable posture label for a score.
    static func postureLabel(for score: Int) -> String {
        switch score {
        case 80...100: return "Robust"
        case 50...79:  return "Baseline"
        case 20...49:  return "Vulnerable"
        default:       return "Fragile"
        }
    }
}
```

> **Note:** Individual finding scores are assigned by each scanner at creation time, not here. Scanners know why a finding is risky; ScoringEngine aggregates. This keeps scoring logic co-located with detection.

**Full scoring table** (matches v1 spec + extensions):

| Finding | Status | Score |
|---------|--------|-------|
| Weak algorithm (RSA 1024 / SHA1) | critical | -20 |
| Expired certificate | critical | -15 |
| Cert expiring within 30 days | warn | -10 |
| Unknown/unexpected identity | critical | -25 |
| Known identity (developer / TLS / SSH) | warn | -5 |
| mkcert root CA | warn | -15 |
| Custom root CA (user trust domain) | warn | -15 |
| World-readable private key | critical | -40 |
| Private key in git repo | critical | -30 |
| Private key in Docker-related path | warn | -15 |
| Any private key file | warn | -10 |
| Docker mount to /dev-pki | critical | -20 |
| Docker mount to /etc/ssl, /certs | warn | -10 |

import Foundation

struct ReportWriter {

    // MARK: - JSON

    static func json(report: ScanReport) throws -> String {
        let encoder = JSONEncoder()
        encoder.outputFormatting     = [.prettyPrinted, .sortedKeys]
        encoder.dateEncodingStrategy = .iso8601

        return String(data: data, encoding: .utf8) ?? ""

    static func markdown(report: ScanReport) -> String {
        let df         = ISO8601DateFormatter()
        let scoreLabel = ScoringEngine.postureLabel(for: report.totalScore)
        var lines: [String] = [
            "# Secure Pride — PKI Trust Audit",
            "",
            "**Date:** \(df.string(from: report.timestamp))  ",
            "**Score:** \(report.totalScore)/100 (\(scoreLabel))  ",
            "**Certificates:** \(report.summary.certificateCount)  ",
            "**Identities:** \(report.summary.identityCount)  ",
            "**Trust Anchors:** \(report.summary.trustAnchorCount)  ",
            "**Filesystem Keys:** \(report.summary.filesystemKeyCount)  ",
            "**Docker Findings:** \(report.summary.dockerFindingCount)  ",
            "**Critical:** \(report.summary.criticalCount)  ",
            "**Warnings:** \(report.summary.warnCount)  ",
            "", "---", "", "## Findings", ""
        ]

        let groups: [(String, [Finding])] = [
            ("🔴 Critical", report.findings.filter { $0.status == .critical }),
            ("🟡 Warning",  report.findings.filter { $0.status == .warn }),
            ("✅ OK",       report.findings.filter { $0.status == .ok }),
        ]

        for (header, findings) in groups where !findings.isEmpty {
            lines.append("### \(header)")
            lines.append("")
            for f in findings {
                lines += [
                    "#### \(f.id) — \(f.name)", "",
                    f.detail, ""
                ]
                if !f.metadata.isEmpty {
                    lines += ["| Key | Value |", "|-----|-------|"]
                    for (k, v) in f.metadata.sorted(by: { $0.key < $1.key }) {
                        lines.append("| \(k) | \(v) |")
                    }
                    lines.append("")
                }
            }
        }

        if !report.errors.isEmpty {
            lines += ["---", "", "## Scan Errors", ""]
            for e in report.errors {
                lines.append("- **\(e.scanner):** \(e.message) (code: \(e.code))")
            }
        }

        return lines.joined(separator: "\n")
    }

    // MARK: - Text

    static func text(report: ScanReport) -> String {
        let label = ScoringEngine.postureLabel(for: report.totalScore)
        var lines = [
            "secure-pride audit — PKI trust inventory",

            "Found: \(report.summary.certificateCount) certs, \(report.summary.identityCount) identities, \(report.summary.trustAnchorCount) trust anchors",
            "       \(report.summary.filesystemKeyCount) filesystem keys, \(report.summary.dockerFindingCount) docker findings",
            "Critical: \(report.summary.criticalCount)  Warnings: \(report.summary.warnCount)",

        let nonOk = report.findings.filter { $0.status != .ok }
        if nonOk.isEmpty {
            lines.append("No issues found.")
        } else {
            for f in nonOk {
                let prefix = f.status == .critical ? "CRIT" : "WARN"
                lines.append("[\(prefix)] \(f.id) \(f.name)")
                lines.append("       \(f.detail)")
            }
        }

        return lines.joined(separator: "\n")
    }

import Foundation

/// Run all scanners and return a scored ScanReport.
/// - Parameters:
///   - includeDocker: When true, DockerScanner runs in addition to always-on scanners.
///   - category: When non-nil, only the named category is scanned. Nil = all categories.
func runScan(includeDocker: Bool = false, category: String? = nil) -> ScanReport {
    var findings: [Finding]   = []
    var errors:   [ScanError] = []

    // Always-on scanners — gated by category filter
    if category == nil || category == "certificate" {
        let r = CertificateScanner.scan()
        findings += r.findings; errors += r.errors
    }
    if category == nil || category == "identity" {
        let r = IdentityScanner.scan()
        findings += r.findings; errors += r.errors
    }
    if category == nil || category == "trust-setting" {
        let r = TrustScanner.scan()
        findings += r.findings; errors += r.errors
    }
    if category == nil || category == "filesystem" {
        let r = FilesystemScanner.scan()
        findings += r.findings; errors += r.errors
    }

    // Opt-in: Docker
    if includeDocker && (category == nil || category == "docker") {
        let r = DockerScanner.scan()
        findings += r.findings; errors += r.errors
    }

    return ScanReport(
        timestamp:  Date(),
        findings:   findings,
        totalScore: ScoringEngine.totalScore(from: findings),
        summary:    ScoringEngine.summarize(findings: findings),
        errors:     errors,
        trustGraph: GraphBuilder.build(from: findings)
    )
}

// Sources/SecurePride/Server/DashboardServer.swift
import Foundation
import Network

final class DashboardServer {

    private let port: Int
    private var listener: NWListener?
    private var currentReport: ScanReport?
    private let queue = DispatchQueue(label: "secure-pride.server", qos: .userInitiated)
    private let dashboardHTML: String

    init(port: Int) {
        self.port = port

        // Locate Dashboard/index.html
        let execDir = URL(fileURLWithPath: CommandLine.arguments[0]).deletingLastPathComponent()
        let candidates = [
            execDir.appendingPathComponent("Dashboard/index.html"),
            execDir.appendingPathComponent("../Dashboard/index.html"),
            URL(fileURLWithPath: FileManager.default.currentDirectoryPath)
                .appendingPathComponent("Dashboard/index.html"),
        ]

        if let url = candidates.first(where: { FileManager.default.fileExists(atPath: $0.path) }),
           let html = try? String(contentsOf: url, encoding: .utf8) {
            dashboardHTML = html
        } else {
            dashboardHTML = "<html><body><h1>Dashboard not found. Run from repo root.</h1></body></html>"
        }
    }

    func start(with report: ScanReport) {
        currentReport = report
        guard let nwPort = NWEndpoint.Port(rawValue: UInt16(port)) else {
            fputs("Invalid port \(port)\n", stderr); return
        }
        listener = try? NWListener(using: .tcp, on: nwPort)
        listener?.newConnectionHandler = { [weak self] in self?.handleConnection($0) }
        listener?.start(queue: queue)
    }

    // MARK: - Connections

    private func handleConnection(_ connection: NWConnection) {
        connection.start(queue: queue)
        connection.receive(minimumIncompleteLength: 1, maximumLength: 65536) {
            [weak self] data, _, _, _ in
            guard let self, let data, !data.isEmpty else { connection.cancel(); return }
            self.handleRequest(String(data: data, encoding: .utf8) ?? "",
                               connection: connection)
        }
    }

    private func handleRequest(_ request: String, connection: NWConnection) {
        let line  = request.components(separatedBy: "\r\n").first ?? ""
        let parts = line.split(separator: " ")
        guard parts.count >= 2 else {
            return send(connection: connection, status: "400 Bad Request",
                        body: "Bad Request", type: "text/plain")
        }
        let method = String(parts[0])
        let path   = String(parts[1])

        switch (method, path) {
        case ("GET", "/"), ("GET", "/index.html"):
            send(connection: connection, status: "200 OK",
                 body: dashboardHTML, type: "text/html; charset=utf-8")

        case ("GET", "/api/scan.json"):
            send(connection: connection, status: "200 OK",
                 body: encodedReport(), type: "application/json")

        case ("POST", "/api/scan"):
            currentReport = runScan()
            send(connection: connection, status: "200 OK",
                 body: encodedReport(), type: "application/json")

        default:
            send(connection: connection, status: "404 Not Found",
                 body: "Not Found", type: "text/plain")
        }
    }

    private func send(
        connection: NWConnection,
        status: String,
        body: String,
        type contentType: String
    ) {
        let bodyData = body.data(using: .utf8) ?? Data()
        var response = """
            HTTP/1.1 \(status)\r
            Content-Type: \(contentType)\r
            Content-Length: \(bodyData.count)\r
            Connection: close\r
            Cache-Control: no-store\r
            \r

            """.data(using: .utf8)!
        response.append(bodyData)
        connection.send(content: response, completion: .contentProcessed { _ in connection.cancel() })
    }

    private func encodedReport() -> String {
        guard let r = currentReport else { return "{}" }

// ── Trust graph ───────────────────────────────────────────────
function renderGraph(graph) {
  const svg = document.getElementById('trust-graph-svg');
  // Clear SVG content safely
  while (svg.firstChild) svg.removeChild(svg.firstChild);

  if (!graph || !graph.nodes || graph.nodes.length === 0) {
    const t = document.createElementNS(SVG_NS, 'text');
    t.setAttribute('x', '400');
    t.setAttribute('y', '200');
    t.setAttribute('text-anchor', 'middle');
    t.setAttribute('fill', '#A7A9A9');
    t.setAttribute('font-family', 'Inter,sans-serif');
    t.setAttribute('font-size', '14');
    t.textContent = 'No trust chain findings to display.';
    svg.appendChild(t);
    return;
  }

  const W = 800;
  const H = Math.max(400, graph.nodes.length * 90);
  svg.setAttribute('viewBox', '0 0 ' + W + ' ' + H);

  // Build adjacency
  const childrenMap = {};
  const parentMap   = {};
  graph.nodes.forEach(n => { childrenMap[n.id] = []; });
  graph.edges.forEach(e => {
    if (!childrenMap[e.from]) childrenMap[e.from] = [];
    childrenMap[e.from].push(e.to);
    parentMap[e.to] = e.from;
  });

  // BFS depth assignment from roots
  const roots = graph.nodes.filter(n => !parentMap[n.id]);
  const depth = {};
  const queue = roots.map(n => ({ id: n.id, d: 0 }));
  while (queue.length) {
    const { id, d } = queue.shift();
    if (depth[id] !== undefined) continue;
    depth[id] = d;
    (childrenMap[id] || []).forEach(cid => queue.push({ id: cid, d: d + 1 }));
  }

  // Group by depth
  const byDepth = {};
  graph.nodes.forEach(n => {
    const d = depth[n.id] !== undefined ? depth[n.id] : 0;
    if (!byDepth[d]) byDepth[d] = [];
    byDepth[d].push(n);
  });
  const maxDepth  = Math.max(...Object.keys(byDepth).map(Number));
  const depthStep = H / (maxDepth + 2);

  // Position map
  const pos = {};
  Object.keys(byDepth).forEach(d => {
    const nodes = byDepth[d];
    const step  = W / (nodes.length + 1);
    nodes.forEach((n, i) => {
      pos[n.id] = { x: step * (i + 1), y: depthStep * (Number(d) + 1) };
    });
  });

  // Draw edges
  graph.edges.forEach(e => {
    const p = pos[e.from], c = pos[e.to];
    if (!p || !c) return;
    const path = document.createElementNS(SVG_NS, 'path');
    const midY = (p.y + c.y) / 2;
    path.setAttribute('d',
      'M' + p.x + ',' + p.y +
      ' C' + p.x + ',' + midY +
      ' '  + c.x + ',' + midY +
      ' '  + c.x + ',' + c.y
    );
    path.setAttribute('fill',         'none');
    path.setAttribute('stroke',       'rgba(22,224,245,0.25)');
    path.setAttribute('stroke-width', '1.5');
    svg.appendChild(path);
  });

  // Draw nodes
  graph.nodes.forEach(n => {
    const p = pos[n.id];
    if (!p) return;

    const nodeColor = n.status === 'critical' ? '#FF5459'
                    : n.status === 'warn'     ? '#FFD93D'
                    :                           '#A7A9A9';

    const g = document.createElementNS(SVG_NS, 'g');
    g.setAttribute('transform', 'translate(' + p.x + ',' + p.y + ')');
    g.setAttribute('aria-label', n.label + ' (' + n.type + ')');

    // Glow halo
    const glow = document.createElementNS(SVG_NS, 'circle');
    glow.setAttribute('r',       '20');
    glow.setAttribute('fill',    nodeColor);
    glow.setAttribute('opacity', '0.12');
    g.appendChild(glow);

    // Node circle
    const circle = document.createElementNS(SVG_NS, 'circle');
    circle.setAttribute('r',            '14');
    circle.setAttribute('fill',         'rgba(16,22,41,0.9)');
    circle.setAttribute('stroke',       nodeColor);
    circle.setAttribute('stroke-width', '2');
    g.appendChild(circle);

    // Label (name, truncated)
    const labelEl = document.createElementNS(SVG_NS, 'text');
    labelEl.setAttribute('text-anchor',       'middle');
    labelEl.setAttribute('y',                 '26');
    labelEl.setAttribute('font-size',         '10');
    labelEl.setAttribute('font-family',       'Inter,sans-serif');
    labelEl.setAttribute('fill',              '#FFFFFF');
    labelEl.textContent = (n.label || '').length > 22
      ? n.label.slice(0, 21) + '\u2026'
      : n.label;
    g.appendChild(labelEl);

    // ID label
    const idEl = document.createElementNS(SVG_NS, 'text');
    idEl.setAttribute('text-anchor', 'middle');
    idEl.setAttribute('y',           '38');
    idEl.setAttribute('font-size',   '9');
    idEl.setAttribute('font-family', 'monospace');
    idEl.setAttribute('fill',        '#A7A9A9');
    idEl.textContent = n.id;
    g.appendChild(idEl);

    svg.appendChild(g);
  });
}

import XCTest
@testable import secure_pride

final class GraphBuilderTests: XCTestCase {

        XCTAssertEqual(GraphBuilder.build(from: []), .empty)
    }

    func test_okOnly_returnsEmptyGraph() {
        let graph = GraphBuilder.build(from: [cert("C1", "Root", issuer: nil, status: .ok)])
        XCTAssertTrue(graph.nodes.isEmpty)
    }

    func test_singleWarn_unknownIssuer_createsSyntheticRoot() {
        let graph = GraphBuilder.build(from: [
            cert("C1", "Bad Cert", issuer: "Unknown CA", status: .warn, score: -10)
        ])
        XCTAssertEqual(graph.nodes.count, 2)
        XCTAssertEqual(graph.edges.count, 1)
        XCTAssertNotNil(graph.nodes.first { $0.type == .synthetic })
        XCTAssertEqual(graph.edges.first?.to, "C1")
    }

    func test_selfSigned_noEdge() {
        let graph = GraphBuilder.build(from: [
            cert("C1", "Root CA", issuer: nil, status: .warn, score: -15)
        ])
        XCTAssertEqual(graph.nodes.count, 1)
        XCTAssertTrue(graph.edges.isEmpty)
        XCTAssertEqual(graph.nodes.first?.type, .root)
    }

    func test_twoLevel_chain() {
        // C1 (ok) is included even though it has ok status because it is pulled in
        // via the byName issuer-resolution path when C2's issuer matches C1's name.
        // Only C2 is in `chainable` (warn/critical filter), but C1 is in `byName`
        // (all cert/trust findings) and gets added as context when walking the chain.
        let graph = GraphBuilder.build(from: [
            cert("C1", "Root CA",  issuer: nil,       status: .ok,   score: 0),
            cert("C2", "Leaf Cert", issuer: "Root CA", status: .warn, score: -10),
        ])
        XCTAssertEqual(graph.nodes.count, 2)
        XCTAssertEqual(graph.edges.count, 1)
        XCTAssertEqual(graph.edges.first?.from, "C1")
        XCTAssertEqual(graph.edges.first?.to,   "C2")
    }

    func test_threeLevel_chainNodeTypes() {
        let graph = GraphBuilder.build(from: [
            cert("C1", "Root CA",  issuer: nil,        status: .ok,   score: 0),
            cert("C2", "Dev CA",   issuer: "Root CA",  status: .ok,   score: 0),
            cert("C3", "Leaf",     issuer: "Dev CA",   status: .warn, score: -10),
        ])
        XCTAssertEqual(graph.nodes.count, 3)
        XCTAssertEqual(graph.edges.count, 2)
        XCTAssertEqual(nodeType(graph, "C1"), .root)
        XCTAssertEqual(nodeType(graph, "C2"), .intermediate)
        XCTAssertEqual(nodeType(graph, "C3"), .leaf)
    }

    func test_filesystemFindings_excluded() {
        let graph = GraphBuilder.build(from: [
            Finding(id: "FILE_001", category: .filesystemKey, name: "id_rsa",
                    detail: "", status: .critical, score: -40, metadata: [:])
        ])
        XCTAssertTrue(graph.nodes.isEmpty)
    }

    func test_dockerFindings_excluded() {
        let graph = GraphBuilder.build(from: [
            Finding(id: "DOCK_001", category: .dockerMount, name: "Container: foo",
                    detail: "", status: .critical, score: -20, metadata: [:])
        ])
        XCTAssertTrue(graph.nodes.isEmpty)
    }

    // MARK: - Helpers

    private func cert(
        _ id: String, _ name: String,
        issuer: String?, status: Status, score: Int = 0
    ) -> Finding {
        var meta: [String: String] = [:]
        if let issuer { meta["issuer"] = issuer }
        return Finding(id: id, category: .certificate, name: name,
                       detail: "test", status: status, score: score, metadata: meta)
    }

    private func nodeType(_ graph: TrustGraph, _ id: String) -> NodeType? {
        graph.nodes.first { $0.id == id }?.type
    }
}
